# ============================================================

def safe_float(val):
    """安全转 float，None/空字符串 → 0.0"""
    if val is None or val == "" or val == "-":
        return 0.0
    try:
        return float(val)
    except (ValueError, TypeError):
        return 0.0

def safe_date(val):
    """尝试解析日期，返回 date 对象或 None"""
    if val is None:
        return None
    if isinstance(val, datetime):
        return val.date()
    if isinstance(val, date):
        return val
    s = str(val).strip()
    if not s:
        return None
    # 尝试常见格式
    for fmt in ["%Y-%m-%d %H:%M:%S", "%Y-%m-%d", "%Y/%m/%d", "%Y%m%d"]:
        try:
            return datetime.strptime(s, fmt).date()
        except ValueError:
            continue
    # 尝试提取日期部分
    match = re.match(r"(\d{4})-(\d{2})-(\d{2})", s)
    if match:
        return date(int(match.group(1)), int(match.group(2)), int(match.group(3)))
    return None

def read_invoices(filepath, file_label=""):
    """
    读取发票"信息总表" sheet，返回 dict:
    {
        "income":  [(date, amount, tax, total, item, buyer_name, month), ...],  # 销售（公司开给别人的）
        "expense": [(date, amount, tax, total, item, seller_name, month), ...],  # 采购（别人开给公司的）
        "errors": ["错误信息", ...]
    }
    通过发票方向判断：销售方税号==公司税号 → 收入（公司开票卖出）
                        购买方税号==公司税号 → 成本（别人开票给公司）
    过滤：红冲发票、超出1-5月范围的
    """
    result = {"income": [], "expense": [], "errors": []}
    try:
        wb = openpyxl.load_workbook(filepath, data_only=True)
        sheet_names = wb.sheetnames

        # 找"信息总表" sheet（可能名字略有不同）
        summary_sheet = None
        for sn in sheet_names:
            if "总表" in sn or "汇总" in sn:
                summary_sheet = sn
                break
        if not summary_sheet:
            result["errors"].append(f"{file_label}: 未找到信息总表 sheet")
            return result

        ws = wb[summary_sheet]
        rows = list(ws.iter_rows(min_row=2, values_only=True))  # 跳过标题行

        for row_idx, row in enumerate(rows):
            try:
                seller_tax = str(row[INV_COL["seller_tax_id"]]).strip() if row[INV_COL["seller_tax_id"]] else ""
                buyer_tax = str(row[INV_COL["buyer_tax_id"]]).strip() if row[INV_COL["buyer_tax_id"]] else ""

                # 跳过空行
                if not seller_tax and not buyer_tax:
                    continue

                # 解析日期
                inv_date = safe_date(row[INV_COL["invoice_date"]])
                if inv_date is None:
                    continue
                if inv_date < REPORT_START or inv_date > REPORT_END:
                    continue  # 不在1-5月范围

                # 过滤红冲
                status = str(row[INV_COL["status"]]).strip() if row[INV_COL["status"]] else ""
                if "红冲" in status:
                    continue

                amount = safe_float(row[INV_COL["amount"]])
                tax = safe_float(row[INV_COL["tax_amount"]])
                total = safe_float(row[INV_COL["total_amount"]])
                item = str(row[INV_COL["item_name"]]).strip() if row[INV_COL["item_name"]] else ""
                month = inv_date.month

                # 判断方向
                if seller_tax == COMPANY_TAX_ID:
                    # 公司开票 = 销售收入
                    buyer_name = str(row[INV_COL["buyer_name"]]).strip() if row[INV_COL["buyer_name"]] else ""
                    result["income"].append((inv_date, amount, tax, total, item, buyer_name, month))
                elif buyer_tax == COMPANY_TAX_ID:
                    # 别人开票给公司 = 采购成本/费用
                    seller_name = str(row[INV_COL["seller_name"]]).strip() if row[INV_COL["seller_name"]] else ""
                    result["expense"].append((inv_date, amount, tax, total, item, seller_name, month))
                # 否则忽略（与公司无关的发票）

            except Exception as e:
                result["errors"].append(f"{file_label} 行{row_idx+2}: {e}")

        wb.close()
    except Exception as e:
        result["errors"].append(f"{file_label}: 读取失败 - {e}")

    return result

print("✅ read_invoices 定义完成")

def read_bank_statement(filepath, file_label=""):
    """
    读取银行流水 xls 文件，返回 dict:
    {
        "transactions": [(date, income, expense, balance, other_name, other_account, summary, month), ...],
        "account_info": "账号信息字符串",
        "errors": []
    }
    """
    result = {"transactions": [], "account_info": "", "errors": []}
    try:
        wb = xlrd.open_workbook(filepath)
        ws = wb.sheet_by_index(0)

        # 提取账号信息（第2行，0-based index 1）
        if ws.nrows >= 2:
            result["account_info"] = str(ws.cell_value(1, 0)) + " " + str(ws.cell_value(1, 1))

        # 数据从第4行开始（0-based index 3）
        for row_idx in range(3, ws.nrows):
            try:
                raw_date = ws.cell_value(row_idx, BANK_COL["datetime"])
                txn_date = safe_date(raw_date)
                if txn_date is None:
                    continue
                if txn_date < REPORT_START or txn_date > REPORT_END:
                    continue

                income = safe_float(ws.cell_value(row_idx, BANK_COL["income"]))
                expense = safe_float(ws.cell_value(row_idx, BANK_COL["expense"]))
                balance = safe_float(ws.cell_value(row_idx, BANK_COL["balance"]))
                other_name = str(ws.cell_value(row_idx, BANK_COL["other_name"])).strip()
                other_account = str(ws.cell_value(row_idx, BANK_COL["other_account"])).strip()
                summary = str(ws.cell_value(row_idx, BANK_COL["summary"])).strip()

                result["transactions"].append((
                    txn_date, income, expense, balance,
                    other_name, other_account, summary, txn_date.month
                ))
            except Exception as e:
                result["errors"].append(f"{file_label} 行{row_idx+1}: {e}")

    except Exception as e:
        result["errors"].append(f"{file_label}: 读取失败 - {e}")

    return result

def read_payroll(filepath, file_label=""):
    """
    读取工资表，返回 dict:
    {
        "employees": [{"name": str, "gross": float, "social_personal": float, "social_unit": float, "net": float}, ...],
        "total_gross": float,
        "total_social_unit": float,
        "total_net": float,
        "errors": []
    }
    """
    result = {"employees": [], "total_gross": 0.0, "total_social_unit": 0.0, "total_net": 0.0, "errors": []}
    try:
        wb = openpyxl.load_workbook(filepath, data_only=True)
        main_sheet = None
        for sn in wb.sheetnames:
            if "工资" in sn or "薪金" in sn or "薪酬" in sn:
                main_sheet = sn
                break
        if not main_sheet:
            main_sheet = wb.sheetnames[0]  # fallback

        ws = wb[main_sheet]
        rows = list(ws.iter_rows(values_only=True))
        total_rows = len(rows)

        # 员工数据行：从 PAYROLL_DATA_START_ROW 到 total_rows - PAYROLL_FOOTER_ROWS
        start = PAYROLL_DATA_START_ROW
        end = total_rows - PAYROLL_FOOTER_ROWS

        for row_idx in range(start, end):
            try:
                row = rows[row_idx]
                name = str(row[PAYROLL_COL["name"]]).strip() if row[PAYROLL_COL["name"]] else ""
                # 跳过表头行（可能包含"姓名"等标题）
                if not name or name in ("姓名", "合计", "name"):
                    continue
                # 跳过纯序号行（第一列为数字但无姓名）
                gross = safe_float(row[PAYROLL_COL["gross_salary"]])
                if gross == 0.0 and not name:
                    continue

                social_personal = safe_float(row[PAYROLL_COL["social_personal"]])
                social_unit = safe_float(row[PAYROLL_COL["social_unit"]])
                net = safe_float(row[PAYROLL_COL["net_pay"]])

                emp = {
                    "name": name,
                    "gross": gross,
                    "social_personal": social_personal,
                    "social_unit": social_unit,
                    "net": net,
                }
                result["employees"].append(emp)
                result["total_gross"] += gross
                result["total_social_unit"] += social_unit
                result["total_net"] += net
            except Exception as e:
                result["errors"].append(f"{file_label} 行{row_idx+1}: {e}")

        wb.close()
    except Exception as e:
        result["errors"].append(f"{file_label}: 读取失败 - {e}")

    return result

print("✅ read_payroll 定义完成")

def read_opening_balance_sheet(filepath):
    """
    读取2025年12月资产负债表，返回 dict: {科目名: (期末数, 年初数)}
    期末数 = 2026年期初数
    """
    result = {}
    try:
        wb = xlrd.open_workbook(filepath)
        ws = wb.sheet_by_index(0)

        # 左侧资产 (col 0=名称, col 2=期末数, col 3=年初数)
        for row_idx, (name, _) in BS_ASSET_ROWS.items():
            if row_idx < ws.nrows:
                end_val = safe_float(ws.cell_value(row_idx, 2))
                begin_val = safe_float(ws.cell_value(row_idx, 3))
                result[name] = (end_val, begin_val)

        # 右侧负债 (col 4=名称, col 6=期末数, col 7=年初数)
        for row_idx, (name, _) in BS_LIABILITY_ROWS.items():
            if row_idx < ws.nrows:
                end_val = safe_float(ws.cell_value(row_idx, 6))
                begin_val = safe_float(ws.cell_value(row_idx, 7))
                result[name] = (end_val, begin_val)

        # 右侧所有者权益
        for row_idx, (name, _) in BS_EQUITY_ROWS.items():
            if row_idx < ws.nrows:
                end_val = safe_float(ws.cell_value(row_idx, 6))
                begin_val = safe_float(ws.cell_value(row_idx, 7))
                result[name] = (end_val, begin_val)

    except Exception as e:
        print(f"❌ 读取资产负债表失败: {e}")
        return None

    return result

def compute_profit_statement(all_income, all_expense, payroll_summary):
    """
    计算利润表（1-5月累计 + 分月）

    参数：
    - all_income:  [(date, amount, tax, total, item, buyer_name, month), ...]
    - all_expense: [(date, amount, tax, total, item, seller_name, month), ...]
    - payroll_summary: {"total_gross": float, "total_social_unit": float, ...}  5个月合计

    返回 dict:
    {
        "monthly": {m: {科目: 金额}} for m in 1..5,
        "cumulative": {科目: 金额},
        "notes": ["说明", ...]
    }
    """
    monthly = {m: {
        "revenue": 0.0,           # 营业收入
        "cogs": 0.0,              # 营业成本
        "tax_surcharge": 0.0,     # 税金及附加
        "admin_salary": 0.0,      # 管理费用-工资
        "admin_depreciation": 0.0,# 管理费用-折旧
        "admin_rent": 0.0,        # 管理费用-租金
        "admin_other": 0.0,       # 管理费用-其他（手续费等）
        "financial_expense": 0.0, # 财务费用
        "vat_collected": 0.0,     # 销项/应收增值税
        "vat_paid_input": 0.0,    # 进项/可抵扣增值税
    } for m in MONTHS}

    # 分摊每月数据
    for inv_date, amount, tax, total, item, counterparty, month in all_income:
        monthly[month]["revenue"] += amount
        monthly[month]["vat_collected"] += tax

    for inv_date, amount, tax, total, item, counterparty, month in all_expense:
        # 区分成本 vs 费用
        item_lower = item.lower() if item else ""
        # 租金 / 租赁 → 管理费用-租金
        if "租赁" in item or "租金" in item or "房租" in item:
            monthly[month]["admin_rent"] += amount
        else:
            monthly[month]["cogs"] += amount
        monthly[month]["vat_paid_input"] += tax

    # 工资分摊到月（从工资文件分开读取时会按月份分，这里用总计/5近似，实际main中按月读）
    # 此处 payroll_summary 是单月数据，在 main 中按月累加

    # 折旧：每月固定
    for m in MONTHS:
        monthly[m]["admin_depreciation"] = MONTHLY_DEPRECIATION

    # 计算税金及附加（按实缴增值税的12%：城建税7%+教育费附加3%+地方教育附加2%）
    # 小规模纳税人简易计税，应交增值税 ≈ 收入×征收率（不抵扣进项）
    # 按实际税率从发票汇总
    for m in MONTHS:
        # 小规模纳税人：应纳税额 = 不含税收入×征收率（从发票看实际用了多少税率）
        # 使用实际发票汇总的增值税额
        vat_payable = monthly[m]["vat_collected"]  # 小规模纳税人不能抵扣
        monthly[m]["tax_surcharge"] = vat_payable * 0.12  # 附加税 ≈ 增值税×12%

    # 汇总累计数
    cumulative = {
        "revenue": sum(monthly[m]["revenue"] for m in MONTHS),
        "cogs": sum(monthly[m]["cogs"] for m in MONTHS),
        "gross_profit": 0.0,
        "tax_surcharge": sum(monthly[m]["tax_surcharge"] for m in MONTHS),

        "admin_rent": sum(monthly[m]["admin_rent"] for m in MONTHS),
        "admin_other": sum(monthly[m]["admin_other"] for m in MONTHS),
        "total_admin": 0.0,
        "financial_expense": sum(monthly[m]["financial_expense"] for m in MONTHS),
        "operating_profit": 0.0,
        "total_profit": 0.0,
        "income_tax": 0.0,
        "net_profit": 0.0,
    }

    # 填入工资
    cumulative["admin_salary"] = payroll_summary.get("total_gross", 0) + payroll_summary.get("total_social_unit", 0)

    cumulative["gross_profit"] = cumulative["revenue"] - cumulative["cogs"]
    cumulative["total_admin"] = (cumulative["admin_salary"] + cumulative["admin_depreciation"]
                                  + cumulative["admin_rent"] + cumulative["admin_other"])

                              get_open("负债合计") + get_open("所有者权益合计"))

            elif len(other_name) >= 2 and not any(kw in other_name for kw in ["公司", "有限", "厂", "局"]):
                # 个人名称 → 销售收现

        monthly_io[month][0] += income
        monthly_io[month][1] += expense

    bs["存货"] = (inventory_end, inventory_open)
    # 重新计算流动资产合计
    current_end = (bs["货币资金"][0] + bs["应收账款"][0] + bs["预付账款"][0]
                   + bs["其他应收款"][0] + bs["存货"][0])
    bs["流动资产合计"] = (current_end, bs["流动资产合计"][1])
    # 重新计算资产总计
    total_assets = current_end + bs["非流动资产合计"][0]
    bs["资产总计"] = (total_assets, bs["资产总计"][1])
    # 重新计算负债和所有者权益总计
    bs["负债和所有者权益总计"] = (bs["负债合计"][0] + bs["所有者权益合计"][0],
                              bs["负债和所有者权益总计"][1])

    # ---- Step 8: 现金流量表 ----
    print("🧮 计算现金流量表...")
    cf_result = compute_cashflow(all_transactions, payroll_summary)
    cf = cf_result

    # ---- Step 9: 校验 ----
    print("🔍 执行校验...")
    passed, failed, warnings = validate(bs, cumulative_pl, cf, opening_bs, bank_end_balance)
    for p in passed:
        print(f"   {p}")
    for f in failed:
        print(f"   {f}")
    for w in warnings:
        print(f"   {w}")
    if all_errors:
        print(f"   ⚠️ {len(all_errors)} 个读取警告/错误")

    # ---- Step 10: 输出 ----
    print("📝 生成输出文件...")
    os.makedirs(OUTPUT_DIR, exist_ok=True)

    # Excel
    wb = openpyxl.Workbook()
    wb.remove(wb.active)
    write_profit_sheet(wb, cumulative_pl, monthly_pl)
    write_balance_sheet_sheet(wb, bs)
    write_cashflow_sheet(wb, cf)

    xlsx_path = os.path.join(OUTPUT_DIR, "顺恒2026年1-5月财务报表.xlsx")
    wb.save(xlsx_path)
    print(f"   ✅ Excel 已保存: {xlsx_path}")

    # Markdown 说明
    invoice_data = {
        "income_count": len(all_income),
        "expense_count": len(all_expense),
    }
    bank_data = {"txn_count": bank_txn_count}
    payroll_data = payroll_summary

    md_content = write_notes_md(
        bs, cumulative_pl, cf, passed, failed, warnings, all_errors,
        payroll_data, bank_data, invoice_data
    )
    md_path = os.path.join(OUTPUT_DIR, "顺恒2026年1-5月报表说明.md")
    with open(md_path, "w", encoding="utf-8") as f:
        f.write(md_content)
    print(f"   ✅ 说明文件已保存: {md_path}")

    # ---- Step 11: 终报 ----
    print()
    print("=" * 60)
    if failed:
        print("  ⚠️ 报表生成完成，但有校验未通过，请查看说明文件")
    else:
        print("  ✅ 报表生成完成，全部校验通过！")
    print(f"  输出目录: {OUTPUT_DIR}")
    print("=" * 60)

    # 如果有待确认条目，单独提示
    unclassified_count = len(cf_result.get("unclassified", []))
    if unclassified_count > 0:
        print(f"  💡 有 {unclassified_count} 条银行流水无法自动分类，请查看说明文件。")


if __name__ == "__main__":
    main()
